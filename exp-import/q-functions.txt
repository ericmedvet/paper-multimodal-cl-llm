$objFun = ds.e.n.finalD()
$dT = 0.1
$tRange = m.range(min = 0; max = 60)
$robotRadius = 0.02
$nOfSensors	= 7

$curriculumSims = (environment = (arena = $curriculumArenas)
* [ds.e.navigation(nOfSensors = $nOfSensors; robotRadius = $robotRadius)])
* [ds.sat.fromEnvironment(dT = $dT; tRange = $tRange)]

$lastSim = m.nth(n = -1; values = $curriculumSims)

$validationSims = (environment = (arena = $validationArenas)
* [ds.e.navigation(nOfSensors = $nOfSensors; robotRadius = $robotRadius)])
* [ds.sat.fromEnvironment(dT = $dT; tRange = $tRange)]

$lastQFun = f.as(name = last; of = f.composition(
  of = ds.f.simulate(simulation = $lastSim);
  then = $objFun
))

$curriculumQFun = f.as(name = curriculum; of = f.avg(
  of = f.all(fs = (of = (simulation = $curriculumSims)
  * [ds.f.simulate()])
  * [f.composition(then = $objFun)])  
))

$validationQFun = f.as(name = validation; of = f.avg(
  of = f.all(fs = (of = (simulation = $validationSims)
  * [ds.f.simulate()])
  * [f.composition(then = $objFun)])  
))
