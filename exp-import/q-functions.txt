$objFun = ds.e.n.finalD()
$dT = 0.1
$tRange = m.range(min = 0; max = 60)

$currentQFun = f.composition(
  name = current;
  of = ds.f.simulate(
    simulation = ds.sat.fromEnvironment(
      environment = ds.e.navigation(arena = m.nth(n = -1; values = $curriculumArenas));
      dT = $dT;
      tRange = $tRange
    )
  );
  then = $objFun
)

$allQFun = f.avg(
  name = "current+past";
  of = f.all(
    fs = (of = (simulation = (environment = (arena = $curriculumArenas)
      * [ds.e.navigation()])
      * [ds.sat.fromEnvironment(dT = $dT; tRange = $tRange)])
      * [ds.f.simulate()])
      * [f.composition(then = $objFun)]
  )  
)

$valQFun = f.avg(
  name = "current+past";
  of = f.all(
    fs = (of = (simulation = (environment = (arena = $validationArenas)
      * [ds.e.navigation()])
      * [ds.sat.fromEnvironment(dT = $dT; tRange = $tRange)])
      * [ds.f.simulate()])
      * [f.composition(then = $objFun)]
  )  
)
